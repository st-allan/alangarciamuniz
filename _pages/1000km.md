<script type="text/babel">

const BIKES    = ['Canyon Grizl RAW', 'Canyon Aeroad', 'S-Works Crux'];
const GOAL     = 1000;
const DEADLINE = 'end of 2026';
const ROUTE    = "M 20 340 C 30 300, 15 270, 35 240 C 55 210, 70 230, 80 210 C 90 190, 75 165, 95 145 C 115 125, 140 145, 155 125 C 170 105, 155 75, 175 58 C 195 42, 220 55, 240 45 C 260 35, 270 18, 290 20 C 310 22, 315 42, 335 48 C 355 54, 370 42, 385 52 C 400 62, 398 82, 412 90 C 426 98, 445 88, 455 100 C 465 112, 452 135, 460 150 C 468 165, 485 168, 488 185 C 491 202, 475 215, 472 232 C 469 249, 482 265, 476 282 C 470 299, 450 305, 445 322 C 440 339, 452 355, 445 370";

const formatDate = (iso) => {
  const [y, m, d] = iso.split('-');
  const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
  return `${parseInt(d)} ${months[parseInt(m)-1]} ${y}`;
};

// ── Route Counter ─────────────────────────────────────────────
const RouteCounter = ({ totalKm, goal }) => {
  const pathRef = React.useRef(null);
  const [pathLen, setPathLen] = React.useState(null);
  const [dotPos, setDotPos] = React.useState({ x: 20, y: 340 });
  const pct = Math.min(totalKm / goal, 1);

  React.useEffect(() => {
    if (pathRef.current) {
      const l = pathRef.current.getTotalLength();
      setPathLen(l);
    }
  }, []);

  React.useEffect(() => {
    if (pathRef.current && pathLen != null) {
      const pt = pathRef.current.getPointAtLength(pathLen * pct);
      setDotPos({ x: pt.x, y: pt.y });
    }
  }, [pathLen, pct]);

  return (
    <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', gap: '16px', padding: '32px 0 24px' }}>
      <svg width="340" height="310" viewBox="0 0 510 425" style={{ display: 'block' }}>
        <path className="route-track" d={ROUTE} />
        <path ref={pathRef} className="route-fill" d={ROUTE}
          style={pathLen != null ? { strokeDasharray: pathLen, strokeDashoffset: pathLen * (1 - pct) } : { opacity: 0 }} />
        <circle cx="20" cy="340" r="4" fill="hsl(0,0%,78%)" />
        <text x="20" y="365" textAnchor="middle"
          style={{ fontFamily: "'Libre Franklin',sans-serif", fontSize: '18px', fontWeight: 300, fill: 'hsl(0,0%,62%)' }}>start</text>
        {pathLen != null && pct > 0 && (
          <circle className="route-dot" cx={dotPos.x} cy={dotPos.y} r="6" />
        )}
        <circle cx="445" cy="370" r="4" fill="hsl(0,0%,78%)" />
        <text x="438" y="390" textAnchor="middle"
          style={{ fontFamily: "'Libre Franklin',sans-serif", fontSize: '18px', fontWeight: 300, fill: 'hsl(0,0%,62%)' }}>finish</text>
      </svg>

      <div style={{ width: '100%', textAlign: 'center' }}>
        <div style={{ fontWeight: 300, fontSize: '5rem', lineHeight: 1, color: 'hsl(0,0%,10%)', letterSpacing: '-0.03em', marginBottom: '6px' }}>{totalKm}</div>
        <div style={{ fontSize: '0.875rem', color: 'hsl(0,0%,40%)', marginBottom: '20px' }}>of {goal} km &mdash; {DEADLINE}</div>
        <div style={{ width: '100%', maxWidth: '260px', height: '2px', background: 'hsl(0,0%,88%)', borderRadius: '2px', margin: '0 auto 6px' }}>
          <div style={{ height: '100%', width: `${pct * 100}%`, background: 'hsl(0,0%,10%)', borderRadius: '2px', transition: 'width 1.4s cubic-bezier(0.4,0,0.2,1)' }} />
        </div>
        <div style={{ fontSize: '0.78rem', color: 'hsl(0,0%,55%)' }}>
          {Math.round(pct * 100)}% &middot; {goal - totalKm} km to go
        </div>
      </div>
    </div>
  );
};

// ── Ride Card ─────────────────────────────────────────────────
const RideCard = ({ ride }) => (
  <div className="ride-card">
    {ride.image && (
      <div className="ride-img-wrap">
        <img className="ride-img" src={ride.image} alt={ride.route} />
      </div>
    )}
    <div style={{ fontWeight: 300, fontSize: '1.05rem', color: 'hsl(0,0%,10%)', lineHeight: 1.3, marginBottom: '4px' }}>{ride.route}</div>
    <div style={{ fontSize: '0.8rem', color: 'hsl(0,0%,45%)', marginBottom: '10px' }}>{formatDate(ride.date)}</div>
    <div style={{ display: 'flex', gap: '6px', flexWrap: 'wrap' }}>
      <span className="chip chip-km">{ride.km} km</span>
      <span className="chip">{ride.bike}</span>
    </div>
  </div>
);

// ── Add Ride Form ─────────────────────────────────────────────
const AddRideForm = ({ onAdd, onClose }) => {
  const today = new Date().toISOString().split('T')[0];
  const [form, setForm] = React.useState({ date: today, route: '', km: '', bike: BIKES[0], image: '' });
  const [preview, setPreview] = React.useState(null);
  const set = (k, v) => setForm(f => ({ ...f, [k]: v }));

  const handleImg = (e) => {
    const file = e.target.files[0];
    if (!file) return;
    const url = URL.createObjectURL(file);
    setPreview(url);
    set('image', url);
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    if (!form.route.trim() || !form.km || isNaN(parseFloat(form.km))) return;
    onAdd({ id: Date.now(), date: form.date, route: form.route.trim(), km: parseFloat(form.km), bike: form.bike, image: form.image || null });
    onClose();
  };

  return (
    <div className="form-overlay" onClick={e => { if (e.target === e.currentTarget) onClose(); }}>
      <div className="form-sheet">
        <h2 style={{ fontWeight: 300, fontSize: '1rem', color: 'hsl(0,0%,10%)', margin: '0 0 24px' }}>log a ride</h2>
        <form onSubmit={handleSubmit}>
          <div className="form-row-grid" style={{ marginBottom: '18px' }}>
            <div>
              <label className="form-label">date</label>
              <input type="date" className="form-input" value={form.date} onChange={e => set('date', e.target.value)} required />
            </div>
            <div>
              <label className="form-label">km</label>
              <input type="number" className="form-input" placeholder="e.g. 42" value={form.km} min="0.1" step="0.1" onChange={e => set('km', e.target.value)} required />
            </div>
          </div>
          <div style={{ marginBottom: '18px' }}>
            <label className="form-label">route</label>
            <input type="text" className="form-input" placeholder="e.g. Ajusco gravel loop" value={form.route} onChange={e => set('route', e.target.value)} required />
          </div>
          <div style={{ marginBottom: '18px' }}>
            <label className="form-label">bike</label>
            <select className="form-select" value={form.bike} onChange={e => set('bike', e.target.value)}>
              {BIKES.map(b => <option key={b} value={b}>{b}</option>)}
            </select>
          </div>
          <div style={{ marginBottom: '18px' }}>
            <label className="form-label">photo</label>
            <input type="file" accept="image/*" className="form-input" style={{ padding: '6px 8px', cursor: 'pointer' }} onChange={handleImg} />
            {preview && <img src={preview} alt="preview" style={{ marginTop: '10px', width: '100%', maxHeight: '140px', objectFit: 'cover', borderRadius: '4px', display: 'block' }} />}
          </div>
          <div style={{ display: 'flex', alignItems: 'center', gap: '4px' }}>
            <button type="submit" className="btn-add">add ride</button>
            <button type="button" className="btn-cancel" onClick={onClose}>cancel</button>
          </div>
        </form>
      </div>
    </div>
  );
};

// ── App ───────────────────────────────────────────────────────
const KmPage = () => {
  // Merge Jekyll-injected rides with any locally-added ones
  const [rides, setRides] = React.useState(() => {
    const base = window.__JEKYLL_RIDES__ || [];
    try {
      const local = JSON.parse(localStorage.getItem('1000km_local_rides') || '[]');
      // Avoid duplicates: local rides have id > base max id
      const baseIds = new Set(base.map(r => r.id));
      const extras = local.filter(r => !baseIds.has(r.id));
      return [...base, ...extras];
    } catch { return base; }
  });
  const [showForm, setShowForm] = React.useState(false);

  const addRide = (ride) => {
    setRides(prev => {
      const next = [ride, ...prev];
      // Persist only locally-added rides (not Jekyll ones)
      const jekyllIds = new Set((window.__JEKYLL_RIDES__ || []).map(r => r.id));
      const local = next.filter(r => !jekyllIds.has(r.id));
      try { localStorage.setItem('1000km_local_rides', JSON.stringify(local)); } catch {}
      return next;
    });
  };

  const totalKm = Math.round(rides.reduce((s, r) => s + r.km, 0) * 10) / 10;
  const sorted = [...rides].sort((a, b) => b.date.localeCompare(a.date));

  return (
    <main style={{ maxWidth: '560px', margin: '0 auto', padding: '0 16px 64px', fontFamily: "'Libre Franklin', sans-serif" }}>
      <div style={{ paddingTop: '24px' }}>
        <h1 style={{ fontWeight: 300, fontSize: '1rem', color: 'hsl(0,0%,10%)', margin: '0 0 2px' }}>1000 km</h1>
        <p style={{ fontSize: '0.8rem', color: 'hsl(0,0%,45%)', margin: 0 }}>a goal to ride 1,000 km before the {DEADLINE}.</p>
      </div>

      <RouteCounter totalKm={totalKm} goal={GOAL} />

      <hr style={{ border: 'none', height: '1px', background: 'hsl(0,0%,90%)', margin: '4px 0 24px' }} />

      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', marginBottom: '4px' }}>
        <span style={{ fontSize: '0.78rem', color: 'hsl(0,0%,55%)' }}>{rides.length} {rides.length === 1 ? 'ride' : 'rides'} logged</span>
        <button className="btn-log-ride" onClick={() => setShowForm(true)}>+ log a ride</button>
      </div>

      <div>
        {sorted.map(ride => <RideCard key={ride.id} ride={ride} />)}
        {rides.length === 0 && (
          <p style={{ fontSize: '0.875rem', color: 'hsl(0,0%,55%)', padding: '32px 0', textAlign: 'center' }}>no rides yet.</p>
        )}
      </div>

      {showForm && <AddRideForm onAdd={addRide} onClose={() => setShowForm(false)} />}
    </main>
  );
};

ReactDOM.createRoot(document.getElementById('km-root')).render(<KmPage />);
</script>
